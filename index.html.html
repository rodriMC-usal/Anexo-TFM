Author

Rodrigo Martínez Contreras

Published

Descripción breve:

Script generado para la limpieza de datos, modelizado GLMM multiescala y generación de gráficos para analizar los efectos de los incendios ocurridos entre 2011 y 2023 sobre los atropellos de fauna en la España peninsular.

1. Organización de datos:

#-----------------------------------------------------------------------------------
#1.1: Poner df en formato long para analizar
#-----------------------------------------------------------------------------------

accidentes_tratamiento <- accidentes_tratamiento %>%
  pivot_longer(cols = starts_with("Mes_"), names_to = "Mes", values_to = "Atropellos") %>%
  mutate(
    Periodo = if_else(str_detect(Mes, "Menos"), "Antes", "Despues"),
    Tratamiento = "Impacto"
  )
accidentes_match <- accidentes_match %>%
  pivot_longer(cols = starts_with("Mes_"), names_to = "Mes", values_to = "Atropellos") %>%
  mutate(
    Periodo = if_else(str_detect(Mes, "Menos"), "Antes", "Despues"),
    Tratamiento = "Control_Match"
  )
accidentes_CCAA <- accidentes_CCAA %>%
  pivot_longer(cols = starts_with("Mes_"), names_to = "Mes", values_to = "Atropellos") %>%
  mutate(
    Periodo = if_else(str_detect(Mes, "Menos"), "Antes", "Despues"),
    Tratamiento = "Control_CCAA"
  )
accidentes_CCAA <- bind_rows(
  accidentes_CCAA %>% mutate(buffer = "10km"),
  accidentes_CCAA %>% mutate(buffer = "2km"),
  accidentes_CCAA %>% mutate(buffer = "quemado")
)
accidentes_CCAA <- accidentes_CCAA %>% 
  filter(id_incendio != "707")

#-----------------------------------------------------------------------------------
#1.2. Eliminar incendios con MENOS de 1 atropello
#-----------------------------------------------------------------------------------

conjunto <- bind_rows(accidentes_tratamiento, accidentes_match, accidentes_CCAA)
incendios_validos <- accidentes_tratamiento %>%
  group_by(id_incendio, buffer, Periodo) %>%
  summarise(Total_Atropellos = sum(Atropellos), .groups = "drop") %>%
  pivot_wider(names_from = Periodo, values_from = Total_Atropellos) %>%
  filter(Antes >= 1 | Despues >= 1) %>%
  select(id_incendio, buffer)
conjunto <- conjunto %>%
  inner_join(incendios_validos, by = c("id_incendio", "buffer"))
conjunto %>% 
  group_by(buffer) %>% 
  summarise(N_Incendios_Unicos = n_distinct(id_incendio))

#-----------------------------------------------------------------------------------
#1.3. Construcción de agrupaciones temporales (bloque anual)
#-----------------------------------------------------------------------------------

datos_conjuntos <- conjunto %>%
  mutate(
    Periodo = factor(Periodo, levels = c("Antes", "Despues")),
    Tratamiento = factor(Tratamiento, levels = c("Control_Match", "Control_CCAA", "Impacto"))
  )
matriz_anual <- datos_conjuntos %>%
  group_by(id_incendio, buffer, Tratamiento, Periodo) %>%
  summarise(Atropellos = sum(Atropellos), .groups = "drop")

4. Análisis BACI agrupado por CCAA:

#-----------------------------------------------------------------------------------
# PARÁMETROS
#-----------------------------------------------------------------------------------

BUFFER           <- "2km" # <--- "2km" | "10km" | "quemado"
CONTROL          <- "Match"    # <--- "Match" | "CCAA"
RESOLUCION       <- "anual"
MIN_INCENDIOS    <- 6         # Mínimo de incendios para usar efecto aleatorio

#-----------------------------------------------------------------------------------
#4.1. Buffer 2km, Control = Match
#-----------------------------------------------------------------------------------

datos_base <- switch(RESOLUCION,
                     "anual" = matriz_anual)
datos <- datos_base %>%
  filter(
    buffer      == BUFFER,
    Tratamiento %in% c("Impacto", paste0("Control_", CONTROL))
  ) %>%
  mutate(
    Tratamiento = factor(Tratamiento,
                         levels = c(paste0("Control_", CONTROL), "Impacto")),
    Periodo     = relevel(factor(Periodo), ref = "Antes")
  ) %>%
  left_join(meta, by = "id_incendio")
ccaa_n <- datos %>%
  filter(Tratamiento == "Impacto") %>%
  distinct(id_incendio, CCAA) %>%
  count(CCAA, name = "n_incendios") %>%
  arrange(desc(n_incendios))
print(ccaa_n)
tabla_ccaa <- lapply(ccaa_n$CCAA, function(ccaa_sel) {
  datos_ccaa <- filter(datos, CCAA == ccaa_sel)
  n_inc      <- ccaa_n$n_incendios[ccaa_n$CCAA == ccaa_sel]
  usa_re     <- n_inc >= MIN_INCENDIOS
  formula_mod <- if (usa_re) {
    Atropellos ~ Tratamiento * Periodo + (1 | id_incendio)
  } else {
    Atropellos ~ Tratamiento * Periodo
  }
  modelo <- tryCatch(
    glmmTMB(formula_mod,
            family = nbinom2(link = "log"),
            data   = datos_ccaa),
    error = function(e) NULL
  )
  if (is.null(modelo)) {
    message(sprintf("    [SKIP] %s: modelo no convergió", ccaa_sel))
    return(NULL)
  }
  res <- tryCatch(
    suppressMessages(simulateResiduals(modelo, n = 500, plot = FALSE)),
    error = function(e) NULL
  )
  if (is.null(res)) {
    message(sprintf("    [SKIP] %s: DHARMa falló", ccaa_sel))
    return(NULL)
  }
  t_unif <- testUniformity(res,    plot = FALSE)
  t_disp <- testDispersion(res,    plot = FALSE)
  t_zi   <- testZeroInflation(res, plot = FALSE)
  coefs    <- summary(modelo)$coefficients$cond
  par_baci <- rownames(coefs)[grep("Impacto.*Despues|Despues.*Impacto",
                                   rownames(coefs))]
  baci_est <- if (length(par_baci)) coefs[par_baci, "Estimate"]   else NA
  baci_se  <- if (length(par_baci)) coefs[par_baci, "Std. Error"] else NA
  baci_p   <- if (length(par_baci)) coefs[par_baci, "Pr(>|z|)"]  else NA
  ctx <- datos_ccaa %>%
    filter(Tratamiento == "Impacto") %>%
    group_by(Periodo) %>%
    summarise(total = sum(Atropellos, na.rm = TRUE), .groups = "drop")
  data.frame(
    CCAA                  = ccaa_sel,
    buffer                = BUFFER,
    control               = CONTROL,
    resolucion            = RESOLUCION,
    n_incendios           = n_inc,
    efecto_aleatorio      = usa_re,
    n_obs                 = nrow(datos_ccaa),
    atrop_antes_impacto   = ctx$total[ctx$Periodo == "Antes"],
    atrop_despues_impacto = ctx$total[ctx$Periodo == "Despues"],
    resid_p               = round(t_unif$p.value, 4),
    resid_OK              = t_unif$p.value > 0.05,
    dispersion_ratio      = round(t_disp$statistic, 3),
    dispersion_p          = round(t_disp$p.value, 4),
    dispersion_OK         = t_disp$p.value > 0.05,
    ZI_ratio              = round(t_zi$statistic, 3),
    ZI_p                  = round(t_zi$p.value, 4),
    ZI_OK                 = t_zi$p.value > 0.05,
    todos_OK              = t_unif$p.value > 0.05 &
      t_disp$p.value > 0.05 &
      t_zi$p.value   > 0.05,
    BACI_est              = round(baci_est, 4),
    BACI_se               = round(baci_se,  4),
    BACI_p                = round(baci_p,   4),
    BACI_cambio_pct       = round((exp(baci_est) - 1) * 100, 1),
    BACI_IC_low_pct       = round((exp(baci_est - 1.96 * baci_se) - 1) * 100, 1),
    BACI_IC_high_pct      = round((exp(baci_est + 1.96 * baci_se) - 1) * 100, 1),
    BACI_sig              = !is.na(baci_p) & baci_p < 0.05,
    stringsAsFactors      = FALSE
  )
})
tabla_ccaa <- bind_rows(tabla_ccaa)
tabla_final <- tabla_ccaa %>%
  select(
    `CCAA` = CCAA, 
    `Nº Incendios` = n_incendios,
    `Tests superados` = todos_OK, 
    `BACI (Est.)` = BACI_est, 
    `p-valor` = BACI_p, 
    `Cambio (%)` = BACI_cambio_pct, 
    `Significativo` = BACI_sig
  ) %>%
  mutate(across(where(is.numeric), ~round(., 3)))
tabla_final %>%
  kable(
    format = "markdown", 
    booktabs = TRUE,
  ) %>%
  kable_styling(
    latex_options = c("striped", "hold_position"),
    full_width = FALSE
  )

resultados_individuales_limpios <- datos_validos %>%
  mutate(
    Tratamiento = factor(Tratamiento, levels = c("Control_Match", "Impacto")),
    Periodo     = factor(Periodo, levels = c("Antes", "Despues"))
  ) %>%
  group_by(id_incendio, buffer) %>%
  group_modify(~ {
    datos_inc <- .x
    imp_antes    <- sum(datos_inc$Atropellos[datos_inc$Tratamiento == "Impacto" & datos_inc$Periodo == "Antes"])
    imp_despues  <- sum(datos_inc$Atropellos[datos_inc$Tratamiento == "Impacto" & datos_inc$Periodo == "Despues"])
    ctrl_antes   <- sum(datos_inc$Atropellos[datos_inc$Tratamiento == "Control_Match" & datos_inc$Periodo == "Antes"])
    ctrl_despues <- sum(datos_inc$Atropellos[datos_inc$Tratamiento == "Control_Match" & datos_inc$Periodo == "Despues"])
    mod <- tryCatch(
      suppressWarnings(glm(Atropellos ~ Tratamiento * Periodo, 
                           family = poisson(link = "log"), 
                           data = datos_inc)),
      error = function(e) NULL
    )
    if(is.null(mod)) {
      return(data.frame(
        Ctrl_Antes = ctrl_antes, Ctrl_Despues = ctrl_despues,
        Imp_Antes = imp_antes, Imp_Despues = imp_despues,
        BACI_p = NA, cambio_pct = NA, notas = "Fallo de convergencia"
      ))
    }
    coefs <- summary(mod)$coefficients
    interaccion <- rownames(coefs)[grep(":", rownames(coefs))]
    if(length(interaccion) > 0) {
      baci_est <- coefs[interaccion, "Estimate"]
      baci_p   <- coefs[interaccion, "Pr(>|z|)"]
      return(data.frame(
        Ctrl_Antes   = ctrl_antes, 
        Ctrl_Despues = ctrl_despues,
        Imp_Antes    = imp_antes, 
        Imp_Despues  = imp_despues,
        BACI_p       = round(baci_p, 4), 
        cambio_pct   = round((exp(baci_est) - 1) * 100, 1),
        notas        = "OK"
      ))
    } else {
      return(data.frame(
        Ctrl_Antes = ctrl_antes, Ctrl_Despues = ctrl_despues,
        Imp_Antes = imp_antes, Imp_Despues = imp_despues,
        BACI_p = NA, cambio_pct = NA, notas = "Sin interacción"
      ))
    }
  }) %>%
  ungroup()
resultados_individuales_limpios %>%
